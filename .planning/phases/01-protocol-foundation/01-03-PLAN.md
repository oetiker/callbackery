<objective>
Define WebSocket specification for real-time bidirectional communication.

Purpose: WebSocket enables backend to push events to frontend without polling. Events are triggers (data_changed, session_expired) that cause frontend to re-fetch via REST. This complements the REST API for real-time updates.

Output:
- WebSocket message format specification
- Connection lifecycle (connect, authenticate, subscribe, disconnect)
- Event subscription mechanism
- Ping/pong keep-alive
</objective>

  <action>
Create WebSocket specification document.

Note: OpenAPI 3.1 doesn't natively support WebSocket, so we document it as an AsyncAPI-style extension or supplementary spec. Structure it clearly for developers.

**Document structure:**

```yaml
# WebSocket Specification for CallBackery Protocol
# Endpoint: ws://host/api/v1/ws (or wss:// for TLS)

websocket:
  endpoint: /api/v1/ws
  description: |
    Real-time bidirectional communication channel.
    Events are triggers that cause frontend to re-fetch via REST.

    Connection flow:
    1. Connect to endpoint
    2. Send authenticate message with token
    3. Receive authenticated confirmation
    4. Subscribe to event types
    5. Receive events, respond with ping/pong
    6. Disconnect when done

messages:
  # Client -> Server messages

  authenticate:
    direction: client-to-server
    description: Authenticate WebSocket connection
    schema:
      type: object
      required: [type, token]
      properties:
        type:
          const: authenticate
        token:
          type: string
          description: JWT token from REST authentication
    example:
      type: authenticate
      token: "eyJhbGciOiJIUzI1NiIs..."

  subscribe:
    direction: client-to-server
    description: Subscribe to event types
    schema:
      type: object
      required: [type, events]
      properties:
        type:
          const: subscribe
        events:
          type: array
          items:
            type: string
          description: Event types to receive
          examples:
            - ["data_changed", "config_changed", "notification"]
        surfaceId:
          type: string
          description: Optional - only events for this surface
    example:
      type: subscribe
      events: ["data_changed", "notification"]

  unsubscribe:
    direction: client-to-server
    description: Unsubscribe from event types
    schema:
      type: object
      required: [type, events]
      properties:
        type:
          const: unsubscribe
        events:
          type: array
          items:
            type: string

  ping:
    direction: client-to-server
    description: Keep-alive ping
    schema:
      type: object
      required: [type]
      properties:
        type:
          const: ping
        timestamp:
          type: string
          format: date-time

  # Server -> Client messages

  authenticated:
    direction: server-to-client
    description: Confirmation of successful authentication
    schema:
      type: object
      required: [type, success]
      properties:
        type:
          const: authenticated
        success:
          type: boolean
        userId:
          type: string
          description: Authenticated user ID
        error:
          type: string
          description: Error message if success is false

  subscribed:
    direction: server-to-client
    description: Confirmation of subscription
    schema:
      type: object
      required: [type, events]
      properties:
        type:
          const: subscribed
        events:
          type: array
          items:
            type: string
          description: Events now subscribed to

  event:
    direction: server-to-client
    description: Push event notification
    schema:
      $ref: './schemas/event.yaml#/components/schemas/Event'
    examples:
      data_changed:
        type: data_changed
        surfaceId: main
        payload:
          pluginId: users-table
        timestamp: "2026-01-23T12:00:00Z"
      notification:
        type: notification
        payload:
          level: success
          message: "Record saved"
        timestamp: "2026-01-23T12:00:00Z"
      session_expired:
        type: session_expired
        payload: {}
        timestamp: "2026-01-23T12:00:00Z"

  pong:
    direction: server-to-client
    description: Keep-alive pong response
    schema:
      type: object
      required: [type]
      properties:
        type:
          const: pong
        timestamp:
          type: string
          format: date-time

  error:
    direction: server-to-client
    description: Error message
    schema:
      type: object
      required: [type, code, message]
      properties:
        type:
          const: error
        code:
          type: string
          enum: [UNAUTHORIZED, INVALID_MESSAGE, SUBSCRIPTION_FAILED]
        message:
          type: string

lifecycle:
  connect:
    description: |
      Client connects to WebSocket endpoint.
      Connection is not authenticated until authenticate message sent.
    timeout: 30s

  authenticate:
    description: |
      Client must send authenticate message within timeout.
      Use same JWT token as REST API authentication.
      Server responds with authenticated message.
    timeout: 10s

  subscribe:
    description: |
      After authentication, client subscribes to event types.
      Can subscribe multiple times to add events.
      Server confirms with subscribed message.

  heartbeat:
    description: |
      Client sends ping, server responds with pong.
      Recommended interval: 30 seconds.
      Server may disconnect if no ping for 90 seconds.
    interval: 30s
    timeout: 90s

  disconnect:
    description: |
      Either side can close the connection.
      Client should clean up subscriptions.
      Reconnect with full authentication flow.
```

```yaml
event_patterns:
  data_changed:
    description: |
      Backend signals that data has changed for a plugin.
      Frontend should re-fetch data via REST.
    trigger: Backend data mutation (create, update, delete)
    frontend_action: |
      1. Check if event.surfaceId matches active surface
      2. Check if event.payload.pluginId matches current plugin
      3. If yes, call GET /config/{pluginId} or refresh data
    example:
      type: data_changed
      surfaceId: main
      payload:
        pluginId: users-table
        action: created
      timestamp: "2026-01-23T12:00:00Z"

  config_changed:
    description: |
      Backend signals that plugin configuration changed.
      Frontend should re-request configuration.
    trigger: Admin changed plugin settings, user permissions changed
    frontend_action: |
      1. Call GET /config/{pluginId}
      2. Re-render with new component tree
    example:
      type: config_changed
      payload:
        pluginId: users-table
      timestamp: "2026-01-23T12:00:00Z"

  session_expired:
    description: |
      User session has expired or was invalidated.
      Frontend should redirect to login.
    trigger: Token expired, user logged out elsewhere, admin revoked session
    frontend_action: |
      1. Clear local auth state
      2. Close WebSocket connection
      3. Redirect to login page
    example:
      type: session_expired
      payload: {}
      timestamp: "2026-01-23T12:00:00Z"

  notification:
    description: |
      Display a notification to the user.
      Frontend shows toast/snackbar.
    trigger: Backend wants to inform user of something
    frontend_action: |
      1. Parse payload as Notification schema
      2. Display toast with level, message, title
      3. Auto-dismiss after duration (if specified)
    example:
      type: notification
      payload:
        level: success
        message: "Record saved successfully"
        title: "Saved"
        duration: 3000
      timestamp: "2026-01-23T12:00:00Z"

  progress:
    description: |
      Report progress of a long-running operation.
      Frontend updates progress indicator.
    trigger: Backend processing large operation
    frontend_action: |
      1. Find progress component by operationId
      2. Update progress percentage
      3. Clear on completion (percent: 100)
    example:
      type: progress
      payload:
        operationId: export-123
        percent: 45
        message: "Processing row 450 of 1000"
      timestamp: "2026-01-23T12:00:00Z"

best_practices:
  - name: Events are triggers
    description: |
      Events should be small and trigger REST calls.
      Don't send large data payloads via WebSocket.
      Frontend re-fetches via REST for data consistency.

  - name: Idempotent handling
    description: |
      Frontend should handle duplicate events gracefully.
      Network issues may cause event redelivery.

  - name: Reconnection
    description: |
      Frontend should auto-reconnect on disconnect.
      Use exponential backoff: 1s, 2s, 4s, 8s, max 30s.
      Re-authenticate and re-subscribe after reconnect.

  - name: Surface scoping
    description: |
      Subscribe with surfaceId to receive only relevant events.
      Reduces noise and processing on frontend.
```
